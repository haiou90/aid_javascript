# 时间倒计时（每秒自动刷新一次）：

```javascript
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>时间日期倒计时</title>
</head>

<body>
    <h1 id="timeBox">
        00:00:00
    </h1>

    <script>
        // 时间补零
        function addZero(num) {
            return num < 10 ? '0' + parseInt(num) : num + ''

        }

        var h1 = document.getElementById('timeBox')
        var timer = setInterval(function () {
            var now = new Date();
            var end = new Date('2020/07/31 12:00:00')
            var leftTime = (end.getTime() - now.getTime()) / 1000
            if (leftTime <= 0) {
                clearInterval(timer)
                h1.innerHTML = '倒计时已结束'
                return
            }
            var H = addZero(parseInt(leftTime / 3600))
            var M = addZero(parseInt(leftTime % 3600 / 60))
            var S = addZero(parseInt(leftTime % 60))
            var time = `${H}:${M}:${S}`
            h1.innerHTML = time

        }, 1000)
    </script>
</body>

</html>
```

# 打乱数组顺序，并获得不重复的值：

```javascript
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>从数组获取不重复的值</title>
</head>

<body>
    <script>
        function getRandElements(arr, len) {
            var res = [];
            // 打乱数组的顺序
            // sort会根据传入的函数的返回值交换两个值的位置
            // Math.random()和0.5运算获取随机的正数或者负数
            // 如果是正数 sort会交换位置
            arr.sort(function () {
                return 0.5 - Math.random()
            })
            // console.log(arr)
            // 获取数组前面len个值
            for (var i = 0; i < len; i++) {
                res.push(arr[i])
            }
            return res
        }
        getRandElements([0, 1, 2, 3, 4, 5, 6], 5)
    </script>
</body>

</html>
```

            // alert(key.value);

            // 检测页面文本中是否有符合正则表达式匹配模式的内容

            // 将用户输入的内容找到以后替换成带有span标签的字符串
            // 'sit' ==> '<span>sit</span>'
            // str.replace()
            var res = str.replace(reg,'<span>'+key.value+'</span>');

            p.innerHTML = res;
        }
